  --spacing-xl: 2rem;
  --spacing-2xl: 3rem;

  /* Border radius */
  --radius-sm: 0.375rem;
  --radius-md: 0.5rem;
  --radius-lg: 0.75rem;

  /* Shadows */

  --shadow-md: 0 4px 6px rgba(0, 0, 0, 0.07);
  --shadow-lg: 0 10px 15px rgba(0, 0, 0, 0.1);
  --shadow-glass: 0 8px 32px rgba(0, 0, 0, 0.1);
}

/* Dark mode overrides */
@variant dark (&:where(.dark, .dark *)) {
  --color-primary-500: #818cf8;
  --color-primary-600: #6366f1;
  --color-primary-400: #a5b4fc;

  --color-surface-1: #1a1a2e;
  --color-surface-2: #252540;

  --shadow-glass: 0 8px 32px rgba(0, 0, 0, 0.3);
}

/* Base layer styles */
@layer base {
  html {
    scroll-behavior: smooth;
  }

  body {
    @apply bg-surface-0 text-text-primary;
    @apply antialiased;
    font-feature-settings: "cv11", "ss01";
  }

  /* Selection styling */
  ::selection {
    @apply bg-primary-500/30;
  }

  }

    @apply bg-clip-text text-transparent;
    background-image: linear-gradient(135deg, var(--color-primary-500), var(--color-primary-400));
  }
